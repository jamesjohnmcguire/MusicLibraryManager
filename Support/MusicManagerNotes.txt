	Besides typical media tags, what other type of data might be included in the metadata?
		Beyond the common media tags (like title, artist, album, year), M4A files (which use the MPEG-4 container format) can contain quite a variety of metadata. Here are the key categories of metadata you might find:

		Technical Audio Information:
				Encoder information and settings
				Original bit rate and sample rate
				Encoding quality parameters
				Normalization settings
				Channel configuration
				Original file format (if converted)
				Compression settings
				Device/Software Information:
				Recording device model
				Software used to create/modify the file
				Software version information
				Encoding/transcoding history
				Creation and modification timestamps
				Device-specific identifiers
			Rights Management:
				Digital Rights Management (DRM) information
				Copyright information
				Purchase information (if from digital store)
				Licensing details
				Content ID markers
			Geographic/Location Data:
				GPS coordinates (if recorded on mobile device)
				Location names
				Recording venue information
				Country codes
			Processing History:
				Audio processing applied
				Filters used
				Volume normalization history
				Mixing/mastering information
				Edit history
			Extended Technical Details:
				Replay gain information
				Peak amplitude values
				Dynamic range information
				Sound check data (iTunes specific)
				Gapless playback information
			Platform-Specific Data:
				iTunes-specific tags
				Platform-specific identifiers
				Store-specific metadata
			Application-specific settings
			File System Metadata:
				Original filename
				File creation date
				Last modified date
				File permissions
				Operating system attributes

	caf
		.caf (Core Audio Format) is Apple's proprietary audio file format developed for macOS and iOS.

		Key Characteristics of .caf:
			Advantages:
				Massive file support - Can store files larger than 4GB (unlike WAV's 2GB limit)
				Extremely flexible - Can contain virtually any audio codec (PCM, ALAC, AAC, MP3, etc.)
				Streamable - Designed for efficient streaming
				Extensive metadata - Can store rich metadata and edit information
				Native macOS/iOS support - Core Audio framework handles it natively
		Same container - Still using .m4a (or you could use .caf which is also common for ALAC)

		public class FileOrganizer
		{
			public static void OrganizeFile(string sourceFile, string destinationPattern, TagLib.File tagFile)
			{
				// Replace placeholders in the pattern with tag values.
				string destinationPath = destinationPattern
					.Replace("%artist%", tagFile.Tag.Artists.FirstOrDefault() ?? "Unknown Artist")
					.Replace("%album%", tagFile.Tag.Album ?? "Unknown Album")
					.Replace("%track%", tagFile.Tag.Track.ToString("D2"))
					.Replace("%title%", tagFile.Tag.Title ?? "Unknown Title");

				// Ensure the directory exists.
				var directory = Path.GetDirectoryName(destinationPath);
				Directory.CreateDirectory(directory);

				// Move the file.
				File.Move(sourceFile, destinationPath);
			}
		}

			public class SafeFileOperation
			{
				private readonly List<Operation> _operations = new();

				public async Task<OperationResult> OrganizeFileAsync(
					string sourcePath,
					string template,
					bool previewOnly = true)
				{
					var source = new FileInfo(sourcePath);
					var tags = await ReadTagsAsync(sourcePath);

					var destination = GeneratePath(template, tags);
					var op = new FileOperation
					{
						Source = source.FullName,
						Destination = destination,
						BackupPath = await CreateBackupAsync(source)
					};

					_operations.Add(op);

					if (!previewOnly)
					{
						await ExecuteOperationAsync(op);
					}

					return new OperationResult(op, previewOnly);
				}

				public async Task UndoLastAsync()
				{
					var lastOp = _operations.LastOrDefault();
					if (lastOp != null)
					{
						await RestoreFromBackupAsync(lastOp);
					}
				}
			}
		2. Template Engine for Paths:
			public class PathTemplateEngine
			{
				// Like beets: $artist/$album/$track $title
				public string GeneratePath(string template, AudioMetadata metadata)
				{
					return template
						.Replace("$artist", SafeFileName(metadata.Artist))
						.Replace("$album", SafeFileName(metadata.Album))
						.Replace("$track", metadata.TrackNumber.ToString("D2"))
						.Replace("$title", SafeFileName(metadata.Title));
				}

				private string SafeFileName(string input)
				{
					// Remove invalid chars, trim, etc.
					return Path.GetInvalidFileNameChars()
						.Aggregate(input, (current, c) => current.Replace(c, '_'));
				}
			}
